    # 工单侧补清除时间（SQL更新清除时间）
    
    替换脚本中的参数后执行
    
    ![image.png](https://alidocs.oss-cn-zhangjiakou.aliyuncs.com/res/Q35O851P0m8ZWl9V/img/edc52e40-1f81-4314-b59f-cd4f5c5561c9.png)
    
    ```plsql
    DECLARE
    v_event_fp VARCHAR(100);
    v_sheet_id VARCHAR(100);
    v_cancel_time TIMESTAMP;
    v_cancel_collect_time TIMESTAMP;
    begin
    set v_event_fp := '4049077035_1101744044_2978306659_2899817007_2';  --事件FP
    set v_sheet_id := 'FJ-076-20250921-2760';  --工单号
    set v_cancel_time := '2025-10-01 18:13:13';   --清除时间
    set v_cancel_collect_time := '2025-10-01 18:13:13';  --清除发现时间
    
    --更新主单清除时间
    update
    MONITORWO.MW_ORDER_WORK w
    set w.EVENT_CLEAN_TIME = v_cancel_time
    where W.CURRENTLY_ALARM_FP = v_event_fp;
    
     --更新子单清除时间
    update  "MONITORWO"."MW_ORDER_PUBLIC_INCIDENT" P
    set p.EVENT_CLEAN_TIME =v_cancel_time, p.CLEAR_COLLECTION_TIME=v_cancel_collect_time
    where P.CURRENTLY_ALARM_FP = v_event_fp;
    
    --更新主表的子单信息
      update (
      select W.SHEET_ID,'[' || WM_CONCAT('{"eventNumber":"'|| p.EVENT_NUMBER || '","sheetId":"' ||
      w.SHEET_ID || '"' 
      || case when p.EVENT_CLEAN_TIME is not null then ',"eventCleanTime":"' || p.EVENT_CLEAN_TIME || '"' END
      || case when p.FORCE_CLEAN_TIME is not null then ',"forceCleanTime":"' || p.FORCE_CLEAN_TIME || '"' END
      || ',"isClean":"' || 
      case when p.EVENT_CLEAN_TIME is not null or p.FORCE_CLEAN_TIME is not null then 'YES"}' ELSE 'NO"}' END) 
      || ']' as LINK_MERGERSON_ORDER_ID
       from MONITORWO.MW_ORDER_WORK w left join MONITORWO.MW_ORDER_WORK_RELATED R1 
       on r1.SHEET_ID = w.SHEET_ID
       left join "MONITORWO"."MW_ORDER_PUBLIC_INCIDENT" P
       on r1.SON_EVENT_NUMBER = p.EVENT_NUMBER
      where w.STATUS NOT in ('VOIDED','EXCEPTION','ARCHIVED','FORCED_ARCHIVING','FORCED_VOIDED')
      and EXISTS (select 1 from MONITORWO.MW_ORDER_WORK_RELATED R where r.SHEET_ID = w.SHEET_ID)
      and W.SHEET_ID = v_sheet_id
      GROUP BY W.SHEET_ID) aa left join MONITORWO.MW_ORDER_WORK ww
      on aa.SHEET_ID = ww.SHEET_ID
      set ww.LINK_MERGERSON_ORDER_ID = aa.LINK_MERGERSON_ORDER_ID;
    COMMIT;
    end;
    ```

    1.  先拿单号查，这个是不是满足追单规则；
        
    2.  如果不满足，就先回复他们不满足追单；

    4.  李光烨确认要合并的我们才给处理

```clojurescript
WITH ordered_logs AS (
    SELECT 
        BEGIN_TIME,
        END_TIME,
        LEAD(BEGIN_TIME) OVER (ORDER BY BEGIN_TIME) AS NEXT_BEGIN_TIME
    FROM "MONITORWO".MAPPING_LOG
    WHERE insert_time BETWEEN '2025-08-01 00:00:00' AND '2025-09-28 11:00:01'
)
SELECT 
    'Call "MONITORWO"."MW_ORDER_EVENT_MAPPING_PROC_TB_TMP"('''
    || TO_CHAR(END_TIME, 'YYYY-MM-DD HH24:MI:SS')
    || ''','''
    || TO_CHAR(NEXT_BEGIN_TIME, 'YYYY-MM-DD HH24:MI:SS')
    || ''');' AS call_sql
FROM ordered_logs
WHERE END_TIME < NEXT_BEGIN_TIME
ORDER BY END_TIME;
```

# 动环保电任务穿测优化表统计sql

```json
WITH param AS (
    SELECT
        TIMESTAMP '2025-12-23 00:00:00' AS start_time,
        TIMESTAMP '2025-12-24 00:00:00' AS end_time
    FROM dual
),

/* ================= 城市维表 ================= */
city_dim AS (
    SELECT DISTINCT EVENT_CITY_NAME AS city_name
    FROM "AID"."MW_AID_EVENT_INFO", param p
    WHERE EVENT_TIME >= p.start_time
      AND EVENT_TIME <  p.end_time
),

/* ================= 签到人员 ================= */
sign_cnt AS (
    SELECT
        b.CITY_NAME,
        COUNT(1) AS sign_cnt
    FROM (
        SELECT DISTINCT user_id
        FROM "AID"."MW_AID_STAFF_SIGN_LOG", param p
        WHERE SIGN_STAT = 0
          AND SIGN_TIME >= p.start_time
          AND SIGN_TIME <  p.end_time
          AND USER_NAME NOT IN (
              '叶芳彪','李光烨','张保全','林菊兰','林子旺','余晨'
          )
    ) a
    LEFT JOIN (
        SELECT
            pu.USER_ACCOUNT,
            pr.CITY_NAME,
            ROW_NUMBER() OVER (
                PARTITION BY pu.USER_ACCOUNT
                ORDER BY pr.CITY_NAME
            ) rn
        FROM MONITORWO.portal_user pu
        JOIN MONITORWO.portal_user_role pur ON pu.id = pur.user_id
        JOIN MONITORWO.portal_role pr ON pur.role_id = pr.id
        WHERE pr.SPECIAL_FIELD = 'T2'
    ) b
      ON a.user_id = b.USER_ACCOUNT
     AND b.rn = 1
    GROUP BY b.CITY_NAME
),

/* ================= 派发任务 ================= */
dispatch_cnt AS (
    SELECT
        EVENT_CITY_NAME,
        COUNT(1) AS dispatch_cnt
    FROM "AID"."MW_AID_EVENT_INFO", param p
    WHERE EVENT_TIME >= p.start_time
      AND EVENT_TIME <  p.end_time
    GROUP BY EVENT_CITY_NAME
),

/* ================= 任务闭环 ================= */
close_cnt AS (
    SELECT
        e.EVENT_CITY_NAME,
        COUNT(1) AS close_cnt
    FROM "AID"."MW_AID_EVENT_INFO" e
    JOIN "AID"."MW_AID_TASK_INFO" t
      ON e.EVENT_FP = t.EVENT_FP
     AND t.TASK_STATE = 99999
    JOIN param p ON 1 = 1
    WHERE e.EVENT_TIME >= p.start_time
      AND e.EVENT_TIME <  p.end_time
      AND t.EVENT_TIME >= p.start_time
      AND t.EVENT_TIME <  p.end_time
    GROUP BY e.EVENT_CITY_NAME
),

/* ================= 延期 / 挂起 ================= */
pause_cnt AS (
    SELECT
        e.EVENT_CITY_NAME,
        COUNT(DISTINCT t.TASK_ID) AS pause_cnt
    FROM "AID"."MW_AID_EVENT_INFO" e
    JOIN "AID"."MW_AID_TASK_INFO" t
      ON e.EVENT_FP = t.EVENT_FP
     AND t.TASK_STATE = 99999
    JOIN "AID"."MW_FLOW_TOKEN_LOG" l
      ON t.TASK_ID = l.TASK_ID
     AND l.STATE_ID IN (10150,10350)
    JOIN param p ON 1 = 1
    WHERE e.EVENT_TIME >= p.start_time
      AND e.EVENT_TIME <  p.end_time
      AND l.LOG_DATE >= p.start_time
      AND l.LOG_DATE <  p.end_time
    GROUP BY e.EVENT_CITY_NAME
),

/* ================= 改派 ================= */
reassign_cnt AS (
    SELECT
        e.EVENT_CITY_NAME,
        COUNT(DISTINCT t.TASK_ID) AS reassign_cnt
    FROM "AID"."MW_AID_EVENT_INFO" e
    JOIN "AID"."MW_AID_TASK_INFO" t
      ON e.EVENT_FP = t.EVENT_FP
     AND t.TASK_STATE = 99999
    JOIN "AID"."MW_FLOW_TOKEN_LOG" l
      ON t.TASK_ID = l.TASK_ID
     AND l.FLOW_LINE_ID IN (10040,10050,10190,10200,10360,10500)
    JOIN param p ON 1 = 1
    WHERE e.EVENT_TIME >= p.start_time
      AND e.EVENT_TIME <  p.end_time
      AND l.LOG_DATE >= p.start_time
      AND l.LOG_DATE <  p.end_time
    GROUP BY e.EVENT_CITY_NAME
)

/* ================= 最终汇总 ================= */
SELECT
    c.city_name               AS 地市,
    NVL(s.sign_cnt, 0)        AS 签到人员数量,
    NVL(d.dispatch_cnt, 0)    AS 派发任务数量,
    NVL(cl.close_cnt, 0)      AS 任务闭环数量,
    NVL(pu.pause_cnt, 0)      AS 延期_挂起数量,
    NVL(r.reassign_cnt, 0)    AS 改派任务数量
FROM city_dim c
LEFT JOIN sign_cnt     s  ON c.city_name = s.CITY_NAME
LEFT JOIN dispatch_cnt d  ON c.city_name = d.EVENT_CITY_NAME
LEFT JOIN close_cnt    cl ON c.city_name = cl.EVENT_CITY_NAME
LEFT JOIN pause_cnt    pu ON c.city_name = pu.EVENT_CITY_NAME
LEFT JOIN reassign_cnt r  ON c.city_name = r.EVENT_CITY_NAME
ORDER BY c.city_name;

```

```plsql
  and W.SHEET_ID = v_sheet_id
  GROUP BY W.SHEET_ID) aa left join MONITORWO.MW_ORDER_WORK ww
```

    ```sql
    SELECT

    FROM
        "AID"."MW_AID_TASK_INFO_ARCHIVE" ti

    WHERE
        ti.TASK_ID IN (

        );
    ```
    
    如果任务已经归档，两张表的表名要加\_ARCHIVE

```sql
SELECT
    ti.TASK_ID,
    ti.SHEET_ID,
    ei.ROOM_CUID,
    ei.ROOM_NAME,
    fl.STATE_ID

    "AID"."MW_AID_TASK_INFO" ti
```
